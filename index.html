<script>

    async function uploadImage() {

        const fileInput = document.getElementById('imageInput');

        const statusDiv = document.getElementById('status');

        

        if (fileInput.files.length === 0) {

            alert("Please select an image first!");

            return;

        }


        const formData = new FormData();

        formData.append('image', fileInput.files[0]);


        statusDiv.innerText = "Scanning...";

        statusDiv.style.color = "#666";


        try {

            // Send to your local Python server

            const response = await fetch('http://127.0.0.1:5000/collect', {

                method: 'POST',

                body: formData

            });


            if (response.ok) {

                statusDiv.innerText = "Scan Complete! No leaks found.";

                statusDiv.style.color = "green";

            } else {

                statusDiv.innerText = "Error uploading image.";

                statusDiv.style.color = "red";

            }

        } catch (error) {

            console.error(error);

            statusDiv.innerText = "Server connection failed. Is Python running?";

            statusDiv.style.color = "orange";

        }

    }

</script>
